    private void OK_buttonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_OK_buttonActionPerformed
       char c ,n ,d ;
       MID_num =Integer.parseInt(Manager_ID.getText());
       Manager M1 = new Manager () ;
       c = M1.MSet_ID(MID_num);
       
       if ( c == 'k'){
       if ("ID Found".equals(M1.MCheck_ID())){
           int Door_Num = (Select_Door.getSelectedIndex()) + 1;
           int New_Pass = Integer.parseInt (New_Password.getText());
           int Confirm_Pass = Integer.parseInt (Confirm_Password.getText());
           Update_Door D1 = new Update_Door();
           n= D1.Set_NewPass(New_Pass);
           d = D1.Set_ConfirmPass(Confirm_Pass);
           if(n == 'k' && d == 'k'){
           if (D1.Matched() == true){
            D1.Set_DoorNum(Door_Num);
           
             JOptionPane.showMessageDialog(null, "Password is upadated", "Upadted Successfuly" ,1);
       
           }else {
               JOptionPane.showMessageDialog(this, "Passwords doesn't match", "Warning", JOptionPane.ERROR_MESSAGE);
           }
           }
           else {
              JOptionPane.showMessageDialog(this, "Password must be 5 digits", "Warning", JOptionPane.ERROR_MESSAGE); 
           }
           }
       else {
            JOptionPane.showMessageDialog(this, "Wrong ID", "Warning", JOptionPane.ERROR_MESSAGE);
       }
       }
       else {
           JOptionPane.showMessageDialog(this, "Too large ID", "Warning", JOptionPane.ERROR_MESSAGE);
       }
    }//GEN-LAST:event_OK_buttonActionPerformed
