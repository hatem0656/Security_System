    private void Get_Entrance_NoActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_Get_Entrance_NoActionPerformed
      Services S1 = new Services () ;
      
        switch (Service_State){
                case 'A':
                  queue_no = S1.Asking();
                  break;
                case 'B':
                  queue_no = S1.Booking(); 
                  break;
                case 'C':
                  queue_no = S1.CustomerService();
                  break; 
                default:
                    
        }  
         JOptionPane.showMessageDialog(null, "Your number in queue is " + queue_no, "Queue Number" ,1);
          Customer_GUI g1 = new Customer_GUI();
          this.dispose();
          g1.setVisible(true);
          
    }//GEN-LAST:event_Get_Entrance_NoActionPerformed
